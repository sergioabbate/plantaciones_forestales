def crear_dataframe_desplazado(datos, janela_in, janela_out, column_date, column_y, nome_date = 'date_ref_'):
    df = pd.DataFrame()

    for i in range( len(datos) - janela_in - janela_out+1):
        columna_m = [f'm_{numero}' for numero in range(janela_in,0,-1) ]
        columna_y = [f'y_{numero}' for numero in range(janela_out)]
        t = pd.DataFrame()
        
        
        t[columna_m] = pd.DataFrame(datos[column_y][i:i+janela_in]).T.reset_index(drop=True)
        t[columna_y] = pd.DataFrame(datos[column_y][i+janela_in:i+janela_in+janela_out]).T.reset_index(drop=True)
        t[nome_date+columna_m[-1]] = datos[column_date][i+janela_in-1]
        df = df.append(t.reset_index(drop=True))
        #df = df.reset_index()

    return df.reset_index(drop=True)
